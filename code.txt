## 2️⃣ Modèle `Document.php`

```php
namespace App\Models;

use Illuminate\Database\Eloquent\Factories\HasFactory;
use Illuminate\Database\Eloquent\Model;

class Document extends Model
{
    use HasFactory;

    protected $fillable = [
        'title',
        'filename_one', 'mime_type_one', 'file_data_one',
        'filename_two', 'mime_type_two', 'file_data_two',
    ];

    public function hasFileOne(): bool
    {
        return !is_null($this->file_data_one);
    }

    public function hasFileTwo(): bool
    {
        return !is_null($this->file_data_two);
    }
}
```

## 3️⃣ Contrôleur `DocumentController.php`

```php
namespace App\Http\Controllers;

use Illuminate\Http\Request;
use App\Models\Document;

class DocumentController extends Controller
{
    public function index()
    {
        $documents = Document::all();
        return view('documents.index', compact('documents'));
    }

    // Créer un document avec un ou deux fichiers
    public function store(Request $request)
    {
        $request->validate([
            'title' => 'nullable|string|max:255',
            'file_one' => 'nullable|file|max:5120',
            'file_two' => 'nullable|file|max:5120',
        ]);

        $data = ['title' => $request->title];

        if ($request->hasFile('file_one')) {
            $file = $request->file('file_one');
            $data['filename_one'] = $file->getClientOriginalName();
            $data['mime_type_one'] = $file->getMimeType();
            $data['file_data_one'] = file_get_contents($file->getRealPath());
        }

        if ($request->hasFile('file_two')) {
            $file = $request->file('file_two');
            $data['filename_two'] = $file->getClientOriginalName();
            $data['mime_type_two'] = $file->getMimeType();
            $data['file_data_two'] = file_get_contents($file->getRealPath());
        }

        Document::create($data);

        return redirect()->back()->with('success', 'Document créé avec succès.');
    }

    // Mettre à jour le titre ou chaque fichier
    public function update(Request $request, Document $document)
    {
        $request->validate([
            'title' => 'sometimes|string|max:255',
            'file_one' => 'nullable|file|max:5120',
            'file_two' => 'nullable|file|max:5120',
        ]);

        if ($request->has('title')) {
            $document->title = $request->title;
        }

        if ($request->hasFile('file_one')) {
            $file = $request->file('file_one');
            $document->filename_one = $file->getClientOriginalName();
            $document->mime_type_one = $file->getMimeType();
            $document->file_data_one = file_get_contents($file->getRealPath());
        }

        if ($request->hasFile('file_two')) {
            $file = $request->file('file_two');
            $document->filename_two = $file->getClientOriginalName();
            $document->mime_type_two = $file->getMimeType();
            $document->file_data_two = file_get_contents($file->getRealPath());
        }

        $document->save();

        return redirect()->back()->with('success', 'Document mis à jour avec succès.');
    }

    public function view(Document $document, $fileNumber)
    {
        if ($fileNumber == 1 && $document->hasFileOne()) {
            return response($document->file_data_one)
                ->header('Content-Type', $document->mime_type_one)
                ->header('Content-Disposition', 'inline; filename="' . $document->filename_one . '"');
        }

        if ($fileNumber == 2 && $document->hasFileTwo()) {
            return response($document->file_data_two)
                ->header('Content-Type', $document->mime_type_two)
                ->header('Content-Disposition', 'inline; filename="' . $document->filename_two . '"');
        }

        abort(404, 'Fichier introuvable.');
    }

    public function download(Document $document, $fileNumber)
    {
        if ($fileNumber == 1 && $document->hasFileOne()) {
            return response($document->file_data_one)
                ->header('Content-Type', $document->mime_type_one)
                ->header('Content-Disposition', 'attachment; filename="' . $document->filename_one . '"');
        }

        if ($fileNumber == 2 && $document->hasFileTwo()) {
            return response($document->file_data_two)
                ->header('Content-Type', $document->mime_type_two)
                ->header('Content-Disposition', 'attachment; filename="' . $document->filename_two . '"');
        }

        abort(404, 'Fichier introuvable.');
    }
}
```

## 4️⃣ Routes

```php
use App\Http\Controllers\DocumentController;

Route::get('/documents', [DocumentController::class, 'index'])->name('documents.index');
Route::post('/documents', [DocumentController::class, 'store'])->name('documents.store');
Route::put('/documents/{document}', [DocumentController::class, 'update'])->name('documents.update');

Route::get('/documents/view/{document}/{fileNumber}', [DocumentController::class, 'view'])->name('documents.view');
Route::get('/documents/download/{document}/{fileNumber}', [DocumentController::class, 'download'])->name('documents.download');
```

### ✅ Fonctionnalités de cette version améliorée

1. Deux fichiers (`file_one`, `file_two`) par document
2. Upload **simultané ou séparé** lors de la création
3. Mise à jour **individuelle ou simultanée** des fichiers
4. Visualisation directe dans le navigateur pour images/PDF
5. Téléchargement direct
6. Tout reste **stocké en base de données**, champs nullable
